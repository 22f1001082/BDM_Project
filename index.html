    SELECT 
        c.id,

        SUM(inv.total - inv.taxes) as total_revenue

    JOIN invoices inv ON inv.customer_id = c.id
    GROUP BY c.id              
""", conn)

category_stats = pd.read_sql_query("""
    SELECT
        p.category,
        COUNT(DISTINCT p.id) as sku_count,
        COALESCE(SUM(ii.quantity), 0) as units_sold,               
        COALESCE(SUM(ii.quantity * ii.price), 0) as revenue
    FROM products p            
    LEFT JOIN invoice_items ii ON ii.product_id = p.id
    GROUP BY p.category        
    ORDER BY revenue DESC
""", conn)

branch_stats = pd.read_sql_query("""
    SELECT
        b.name as branch,
        COUNT(DISTINCT inv.id) as invoices,
        SUM(inv.total - inv.taxes) as revenue,  
        COUNT(DISTINCT inv.customer_id) as customers
    FROM invoices inv          
    JOIN branches b ON b.id = inv.branch_id
    GROUP BY b.name            
""", conn)

conn.close()

def styled_table(cell_data, col_labels, row_labels, title, figsize=(13, 2.2),total_row_label=None, y=None):
    n_rows = len(row_labels)   
    n_cols = len(col_labels)
    fig, ax = plt.subplots(figsize=figsize)     
    ax.axis('off')             

    tbl = ax.table(            
        cellText=cell_data,
        rowLabels=row_labels,
        colLabels=col_labels,
        cellLoc='center',
        rowLoc='center',       
        loc='center',
    )       
                
    tbl.auto_set_font_size(False)
    tbl.set_fontsize(13)
    tbl.scale(1, 2.0)

    # Header row
    for col_idx in range(n_cols):   
        cell = tbl[0, col_idx]
        cell.set_facecolor(DARK_GREEN)
        cell.set_text_props(color='white', fontweight='bold')
        cell.set_edgecolor('white') 

    # Row labels and data cells
    for row_idx in range(1, n_rows + 1):
        is_total = (row_labels[row_idx - 1] == total_row_label)

        label_cell = tbl[row_idx, -1]
        label_cell.set_facecolor(LIGHT_GREEN if is_total else DARK_GREEN)      
        label_cell.set_text_props(  
            color='black' if is_total else 'white',
            fontweight='bold'  
        )       
        label_cell.set_edgecolor('white')
        for col_idx in range(n_cols):  
            data_cell = tbl[row_idx, col_idx]
            data_cell.set_facecolor(LIGHT_GREEN if is_total else 'white')      
            data_cell.set_text_props(fontweight='bold' if is_total else 'normal')
            data_cell.set_edgecolor('#cccccc')  

    if not y:
        title_y = 0.85 if figsize[1] < 3 else 0.92  
    fig.text(0.5, y if y else title_y, title,
             ha='center', va='bottom', 
             fontsize=14, fontweight='bold', fontfamily='Liberation Serif')    

    plt.tight_layout()
    plt.show()

def describe_series(s):            
    s = s.dropna()
    return [                       
        int(s.count()),
        f'{s.mean():,.2f}',
        f'{s.median():,.2f}',      
        f'{s.std():,.2f}',
        f'{s.min():,.2f}',         
        f'{s.max():,.2f}',         
        f'{s.skew():.2f}',
    ]

def summary(data):
    daily   = data.groupby('date')['total'].sum() - data.groupby('date')['taxes'].sum()
    monthly = data.groupby(data['date'].dt.to_period('M'))['total'].sum() - data.groupby(data['date'].dt.to_period('M'))['taxes'].sum()
    return {
        'Annual Sales':    data['total'].sum(), 
        'Annual Revenue': data['total'].sum() - data['taxes'].sum(),
        'Avg Monthly Revenue': monthly.mean(),
        'Avg Daily Revenue':   daily.mean(),
        'Avg Invoice Value': data['total'].mean(),
    } 

rows = {}

import sqlite3 
import pandas as pd
import numpy as np                                            
import matplotlib.pyplot as plt                               
               
DARK_GREEN  = '#2d6a4f'                                       
LIGHT_GREEN = '#d8f3dc'                                       
               
DIO_SLOW_THRESHOLD      = 180  # Option 1: DIO > 180 days     
COVERAGE_SLOW_THRESHOLD = 6    # Option 3: closing stock > 6 months of avg monthly sales                                         
               
conn = sqlite3.connect('data.db')                    
               
# ── Inventory: opening + closing aggregated across all branches ───────────────                                                 
inv = pd.read_sql('''                                         
    SELECT product_id,                                        
        SUM(COALESCE(JSON_EXTRACT(opening_balance, '$.value'), 0)) AS opening_value,                                             
        SUM(COALESCE(JSON_EXTRACT(closing_balance, '$.value'), 0)) AS closing_value                                              
    FROM inventory                                            
    GROUP BY product_id                                       
''', conn)                                     

inv_qty = pd.read_sql('''
    SELECT product_id,
        SUM(COALESCE(JSON_EXTRACT(closing_balance, '$.qty'), 0)) AS closing_qty
    FROM inventory
    GROUP BY product_id
''', conn)


# ── Per-SKU purchases ─────────────────────────────────────────────────────────                                                 
sku_purchases = pd.read_sql('''                               
    SELECT product_id, SUM(total) AS purchases                
    FROM purchase_items                                       
    GROUP BY product_id                                       
''', conn)     
               
# ── Per-SKU sales: total revenue + distinct months active ────────────────────                                                  

sales_activity = pd.read_sql('''
    SELECT ii.product_id,
        SUM(ii.price * ii.quantity)                        AS total_sales,
        SUM(ii.quantity)                                   AS total_qty_sold,
        COUNT(DISTINCT strftime('%Y-%m', i.date))          AS sales_months
    FROM invoice_items ii
    JOIN invoices i ON ii.invoice_id = i.id
    GROUP BY ii.product_id
''', conn)


sold_pids = set(sales_activity['product_id'])                 
               
# ── Dead stock: not in invoice_items, excluding grace period ──────────────────                                                 
# Grace period: products whose FIRST purchase was on/after May 15 2025                                                           
grace_pids = set(pd.read_sql('''                              
    SELECT product_id FROM (                                  
        SELECT pi.product_id, MIN(p.date) AS first_purchase   
        FROM purchase_items pi                                
        JOIN purchases p ON pi.purchase_id = p.id             
        GROUP BY pi.product_id                                
    ) WHERE first_purchase >= '2025-05-15'
''', conn)['product_id'])                                     
               
# ── Products table ────────────────────────────────────────────────────────────                                                 
products = pd.read_sql('SELECT id AS product_id, category FROM products', conn)                                                  
conn.close()   
# ── Build per-SKU frame ───────────────────────────────────────────────────────
sku = inv.merge(sku_purchases,  on='product_id', how='left')
sku = sku.merge(products,       on='product_id', how='left')
sku = sku.merge(sales_activity, on='product_id', how='left')
sku = sku.merge(inv_qty, on='product_id', how='left')
sku['total_qty_sold']  = sku['total_qty_sold'].fillna(0)
sku['closing_qty']     = sku['closing_qty'].fillna(0)


sku['purchases']    = sku['purchases'].fillna(0)              
sku['total_sales']  = sku['total_sales'].fillna(0)            
sku['sales_months'] = sku['sales_months'].fillna(0).astype(int)                                                                  
sku['category']     = sku['category'].fillna('Uncategorised')
               
# COGS and inventory metrics
sku['COGS']          = (sku['opening_value'] + sku['purchases'] - sku['closing_value']).clip(lower=0)
sku['avg_inventory'] = (sku['opening_value'] + sku['closing_value']) / 2
sku['ITR']           = np.where((sku['COGS'] > 0) & (sku['avg_inventory'] > 0),                                                  
                                sku['COGS'] / sku['avg_inventory'], np.nan)                                                      
sku['DIO']           = np.where(sku['ITR'] > 0, 365 / sku['ITR'], np.nan)                                                        
               
# Dead stock: no sales in invoice_items AND not in grace period
sku['is_dead']    = (~sku['product_id'].isin(sold_pids)
                     & ~sku['product_id'].isin(grace_pids)
                     & (sku['closing_qty'] > 0))
sku['dead_value'] = np.where(sku['is_dead'], sku['closing_value'], 0)                                                            
 
# Slow-moving: has sales (excludes dead stock)                
has_sales = sku['product_id'].isin(sold_pids)
# Option 1: DIO > threshold
sku['is_slow_dio'] = has_sales & (
    sku['DIO'] > DIO_SLOW_THRESHOLD) & (sku['closing_value'] > 0)
# Option 3: stock coverage > threshold months
# coverage = closing_value / avg_monthly_sales_revenue (at selling price)
sku['avg_monthly_qty'] = np.where(
    sku['sales_months'] > 0,
    sku['total_qty_sold'] / sku['sales_months'],
    np.nan
)
sku['coverage_months'] = np.where(
    (sku['avg_monthly_qty'] > 0) & (sku['closing_qty'] > 0),
    sku['closing_qty'] / sku['avg_monthly_qty'],
    np.nan
)

sku['is_slow_coverage'] = has_sales & (sku['coverage_months'] > COVERAGE_SLOW_THRESHOLD)
               
# ── Aggregate by category ─────────────────────────────────────────────────────
cat = sku.groupby('category', dropna=False).agg(              
    avg_inventory       = ('avg_inventory',     'sum'),
    COGS                = ('COGS',              'sum'),       
    dead_skus           = ('is_dead',           'sum'),
    dead_value          = ('dead_value',        'sum'),       
    slow_dio_skus       = ('is_slow_dio',       'sum'),
    closing_qty         = ('closing_qty',       'sum'),
    total_qty_sold      = ('total_qty_sold',    'sum')
).reset_index()


cat['coverage_months'] = np.where(                            
    cat['total_qty_sold'] > 0,
    cat['closing_qty'] / (cat['total_qty_sold'] / 12),
    np.nan
) 


cat['category'] = cat['category'].fillna('Uncategorised')

cat['DIO'] = np.where(cat['ITR'] > 0, 365 / cat['ITR'], np.nan)                                                                  
cat = cat.sort_values('avg_inventory', ascending=False).reset_index(drop=True)

total_avg_inv = cat['avg_inventory'].sum()

4.1 Figure 13: Inventory Aging by Product Category

ANALYSIS_END = pd.Timestamp('2025-07-15')
BUCKET_ORDER = ['0–3 months', '3–6 months',
                '6–9 months', '9–12 months', '>12 months']
BUCKET_COLORS = ['#52b788', '#b7e4c7', '#ffe066', '#f4a261', '#e63946']

conn = sqlite3.connect('data.db')

# ── Closing qty + opening balance (fallback cost rate for pre-period stock) ───
closing = pd.read_sql('''
    SELECT product_id,                                                                                                           
        SUM(COALESCE(JSON_EXTRACT(closing_balance, '$.qty'),   0)) AS closing_qty,
        SUM(COALESCE(JSON_EXTRACT(opening_balance, '$.qty'),   0)) AS opening_qty,                                               
        SUM(COALESCE(JSON_EXTRACT(opening_balance, '$.value'), 0)) AS opening_value
    FROM inventory                                                                                                               
    GROUP BY product_id
''', conn)

# ── All purchases with dates, newest-first within each product ────────────────
purchases = pd.read_sql('''
    SELECT pi.product_id, p.date, pi.quantity, pi.rate
    FROM purchase_items pi
    JOIN purchases p ON pi.purchase_id = p.id
''', conn)
purchases['date'] = pd.to_datetime(purchases['date'])

products = pd.read_sql('SELECT id AS product_id, category FROM products', conn)
conn.close()

# ── Age bucket classifier ─────────────────────────────────────────────────────


def age_bucket(days):
    if days <= 90:
        return '0–3 months'
    if days <= 180:
        return '3–6 months'
    if days <= 270:
        return '6–9 months'
    if days <= 365:
        return '9–12 months'
    return '>12 months'


# ── Build per-product purchase lookup (sorted newest-first) ───────────────────
purch_groups = {
    pid: grp.sort_values('date', ascending=False)
    for pid, grp in purchases.groupby('product_id')
}

# ── FIFO aging: walk purchases newest → oldest, fill closing qty ──────────────
records = []

for _, row in closing.iterrows():
    pid = row['product_id']
    remaining = row['closing_qty']
    if remaining <= 0:
        continue

    for _, p in purch_groups.get(pid, pd.DataFrame()).iterrows():
        if remaining <= 0:
            break
        take = min(remaining, p['quantity'])
        days = (ANALYSIS_END - p['date']).days
        records.append({
            'product_id': pid,
            'bucket':     age_bucket(days),
            'value':      take * p['rate'],
        })
        remaining -= take

    # Unaccounted remainder → pre-period opening stock
    if remaining > 0:
        op_rate = (row['opening_value'] / row['opening_qty']
                   if row['opening_qty'] > 0 else 0)
        records.append({
            'product_id': pid,
            'bucket':     '>12 months',
            'value':      remaining * op_rate,
        })

# ── Aggregate by category ─────────────────────────────────────────────────────
aging_df = pd.DataFrame(records).merge(products, on='product_id', how='left')
aging_df['category'] = aging_df['category'].fillna('Uncategorised')

cat_aging = (
    aging_df
    .pivot_table(index='category', columns='bucket', values='value', aggfunc='sum')
    .reindex(columns=BUCKET_ORDER)
    .fillna(0)
)
# Sort: largest total at top of chart
cat_aging = cat_aging.loc[cat_aging.sum(
    axis=1).sort_values(ascending=True).index]

# ── Plot ──────────────────────────────────────────────────────────────────────
n_cats = len(cat_aging)
fig, ax = plt.subplots(figsize=(15, n_cats * 0.3 + 2.5))

lefts = np.zeros(n_cats)
for bucket, color in zip(BUCKET_ORDER, BUCKET_COLORS):
    vals = cat_aging[bucket].values / 1e6
    ax.barh(range(n_cats), vals, left=lefts, color=color,
            label=bucket, height=0.5, edgecolor='white')
    label_color = 'black' if color in ('#b7e4c7', '#ffe066') else 'white'
    for i, (v, l) in enumerate(zip(vals, lefts)):
        if v >= 0.4:
            ax.text(l + v / 2, i, f'{v:.1f}M',
                    ha='center', va='center', fontsize=8.5,
                    color=label_color, fontweight='bold')
    lefts += vals
ax.set_yticks(range(n_cats))

4.3 Figure 15: GMROI by Category

DARK_GREEN = '#2d6a4f'
EXCLUDE_CATEGORIES = ['HCV (OTR)', 'PV']

conn = sqlite3.connect('data.db')

cat = pd.read_sql('''
    WITH inv AS (
        SELECT product_id,
            SUM(COALESCE(JSON_EXTRACT(opening_balance, '$.value'), 0)) AS opening_value,
            SUM(COALESCE(JSON_EXTRACT(closing_balance, '$.value'), 0)) AS closing_value
        FROM inventory GROUP BY product_id
    ),
    purch AS (
        SELECT product_id, SUM(total) AS purchases FROM purchase_items GROUP BY product_id
    ),
    sales AS (
        SELECT product_id, SUM(price * quantity) AS revenue FROM invoice_items GROUP BY product_id
    ),
    all_products AS (
        SELECT product_id FROM sales
        UNION
        SELECT product_id FROM inv
    )
    SELECT pr.category,
        COALESCE(SUM(s.revenue), 0)                                                AS revenue,
        SUM(MAX(0, COALESCE(i.opening_value,0) + COALESCE(p.purchases,0)
                   - COALESCE(i.closing_value,0)))                                 AS COGS,
        (SUM(COALESCE(i.opening_value,0)) + SUM(COALESCE(i.closing_value,0))) / 2.0 AS avg_inventory
    FROM all_products ap
    LEFT JOIN inv i   ON ap.product_id = i.product_id
    LEFT JOIN purch p ON ap.product_id = p.product_id
    LEFT JOIN sales s ON ap.product_id = s.product_id
    LEFT JOIN products pr ON ap.product_id = pr.id
    GROUP BY pr.category
''', conn)
conn.close()

cat['category']     = cat['category'].fillna('Uncategorised')
cat['gross_margin'] = cat['revenue'] - cat['COGS']
cat['GMROI']        = np.where(cat['avg_inventory'] > 0,
                          cat['gross_margin'] / cat['avg_inventory'], np.nan)

overall_gmroi = cat['gross_margin'].sum() / cat['avg_inventory'].sum()

cat_active = cat[(cat['revenue'] > 0) & (~cat['category'].isin(EXCLUDE_CATEGORIES))].copy()
c3 = cat_active.dropna(subset=['GMROI']).sort_values('GMROI', ascending=True)

fig, ax = plt.subplots(figsize=(11, len(c3) * 0.4 + 2))
colors = ['#e63946' if v < 0.3 else '#f4a261' if v < 0.6 else DARK_GREEN
          for v in c3['GMROI']]
ax.barh(range(len(c3)), c3['GMROI'], color=colors, height=0.55, edgecolor='white')

for i, v in enumerate(c3['GMROI']):
    ax.text(v + 0.01, i, f'{v:.2f}x', va='center', fontsize=10, fontweight='bold')

ax.axvline(overall_gmroi, color='#333333', linewidth=1.5, linestyle='--',
           label=f'Overall: {overall_gmroi:.2f}x')
ax.set_yticks(range(len(c3)))
ax.set_yticklabels(c3['category'], fontsize=11)
ax.set_xlabel('GMROI  (Gross Margin ÷ Avg Inventory)', fontsize=11)
ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)
ax.set_axisbelow(True)
ax.xaxis.grid(True, alpha=0.3, linestyle='--')

leg = [Patch(facecolor='#e63946', label='< 0.30x  (critical)'),
       Patch(facecolor='#f4a261', label='0.30–0.60x'),
       Patch(facecolor=DARK_GREEN,  label='> 0.60x')]
ax.legend(handles=leg, fontsize=9, loc='lower right')

fig.text(0.5, 1.01, 'Gross Margin Return on Inventory Investment (GMROI) by Category',
         ha='center', fontsize=16, fontweight='bold', fontfamily='Liberation Serif')
fig.text(0.5, 0.97,
         'GMROI = Gross Margin ÷ Avg Inventory  |  Higher = more margin earned per rupee of stock held',
         ha='center', fontsize=9, color='gray', style='italic')
plt.tight_layout()
plt.show()
